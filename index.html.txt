// 初始投入
function saveInitialCapital() {
    initialCapital = parseFloat(document.getElementById('initialCapital').value) || 0;
    saveToCloud();

function updateProgress() {
    let profit = records.filter(r => !r.isDividend).reduce((s, r) => {
        let f = r.type == '收入' ? r.amount + (r.adjustment || 0) : -(r.amount + (r.adjustment || 0));
        return s + f;
    }, 0);
    let pct = initialCapital > 0 ? Math.min(100, Math.max(0, profit / initialCapital * 100)) : 0;
    document.getElementById('capitalProgress').innerHTML = `
        初始投入：${initialCapital.toFixed(2)} 元<br>
        累计经营利润：${profit.toFixed(2)} 元<br>
        回本进度：${pct.toFixed(1)}%
    `;
}

async function addRecord() {

        payDate: document.getElementById('payDate').value,

        channel: document.getElementById('channel').value,
        amount: parseFloat(document.getElementById('amount').value) || 0,

// 加载流水
function loadRecords() {

    b.innerHTML = '';

        tr.innerHTML = `
            <td>${r.payDate}</td>
            <td>${r.belongMonth || '-'}</td>
            <td>${r.type}</td>
            <td>${r.category || '-'}</td>
            <td>${r.channel}</td>
            <td>${r.amount.toFixed(2)}</td>
            <td>${final.toFixed(2)}</td>
            <td>${r.operator}</td>
            <td>${r.remark || ''}</td>`;
        b.appendChild(tr);
    })
}

// 当日对账
function queryDaily() {
    let d = document.getElementById('queryDate').value;
    let list = records.filter(x => x.payDate == d);
    let income = 0, expense = 0;
    let cat = {};
    list.forEach(r => {
        if (r.isDividend) return;
        let f = r.type == '收入' ? r.amount + (r.adjustment || 0) : -(r.amount + (r.adjustment || 0));
        if (r.type == '收入') income += f;
        else { expense += Math.abs(f); cat[r.category] = (cat[r.category] || 0) + Math.abs(f); }
    })
    let s = `${d} 当日对账汇总\n收入：${income.toFixed(2)}\n支出：${expense.toFixed(2)}\n净额：${(income - expense).toFixed(2)}\n\n支出分类：\n`;
    Object.keys(cat).forEach(k => s += `${k}: ${cat[k].toFixed(2)}\n`);
    document.getElementById('summaryResult').innerText = s;
}

// 当日余额
function queryBalance() {
    let d = document.getElementById('queryDate').value;
    let list = records.filter(x => x.payDate <= d);
    let bal = { 支付宝:0,微信:0,兴业银行:0,公司账户:0 };
    list.forEach(r => {
        let f = r.type == '收入' ? r.amount + (r.adjustment || 0) : -(r.amount + (r.adjustment || 0));
        bal[r.channel] += f;
    })
    let s = `截至 ${d} 账户余额\n支付宝：${bal.支付宝.toFixed(2)}\n微信：${bal.微信.toFixed(2)}\n兴业银行：${bal.兴业银行.toFixed(2)}\n公司账户：${bal.公司账户.toFixed(2)}`;
    document.getElementById('summaryResult').innerText = s;
}

    let html=`<h3>${m} 月度汇总</h3><p>收入：${income.toFixed(2)}</p><p>支出：${expense.toFixed(2)}</p><p>净利：${(income-expense).toFixed(2)}</p><table border=1 cellpadding=5 width=100%><tr bgcolor=#eee><th>日期</th><th>归属月</th><th>类型</th><th>分类</th><th>渠道</th><th>金额</th><th>修正</th><th>最终</th><th>操作人</th><th>备注</th></tr>`;
